// include standard library
include std

// prints hello world with newline
std::writeln("Hello, world!")

// define factorial function
define factor(n)
    if n == 1 or n == 0
        return 1
    return n * factor(n - 1)

// call the function
var result = factor(5)
std::writeln(result) // output: 120
